provider "aws" {
  region = var.aws_region
}

# Creates a secure VPC for the migrated environment
resource "aws_vpc" "migration_vpc" {
  cidr_block           = "10.100.0.0/16"
  enable_dns_support   = true
  enable_dns_hostnames = true

  tags = {
    Name        = "migrated-vpc-${var.environment}"
    Project     = "CloudMigration"
    Environment = var.environment
  }
}

# Defines a private subnet for application servers
resource "aws_subnet" "private_subnet_a" {
  vpc_id            = aws_vpc.migration_vpc.id
  cidr_block        = "10.100.1.0/24"
  availability_zone = "${var.aws_region}a"

  tags = {
    Name = "private-subnet-a-${var.environment}"
  }
}
